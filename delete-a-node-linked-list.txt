struct Node
{
	int data;
	Node *next;
};
Node* Insert(Node *head,int x)
{
   Node *temp = new Node();
   temp->data = x;
   temp->next = NULL;
   if(head == NULL) 
   {
       return temp;
   }
   Node *temp1;
   for(temp1 = head;temp1->next!=NULL;temp1= temp1->next);
   temp1->next = temp;return head;
}
/*
  Delete Node at a given position in a linked list 
  Node is defined as 
  struct Node
  {
     int data;
     struct Node *next;
  }
*/
Node* Delete(Node *head, int position)
{
  // Complete this method
    if(position < 0 || head == NULL)
        return head;
    
    int count = 0;
    Node* current = head;
    Node* prev = head;
    
    //Delete Head Node
    if(position == 0)
    {
       head = head -> next;
       return head;
    }
    
    while(current != NULL)
    {
       if(count == position)
       {
          prev->next = current->next;
          current->next = NULL;
          return head;
       }
       
       count++;
       prev = current;
       current = current -> next;
    }
    
    return head;
}

		head=head->next;
